```
kind: pipeline
type: digitalocean
name: greeting

steps:
- name: one
  commands:
  - echo one

- name: two
  commands:
  - echo two

- name: three
  commands:
  - echo three
  depends_on:
  - one
  - two

- name: four
  commands:
  - echo four
  depends_on:
  - three
```
